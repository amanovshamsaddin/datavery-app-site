feature_engineering.py — [SAMPLE] 

import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

def build_features(df: pd.DataFrame) -> pd.DataFrame:
    """
    [SAMPLE FUNCTION]
    Cleans raw data and engineers time-series features for predictive modeling.
    Ensures robust handling of missing values and scales numerical inputs.
    """
    # Create a copy to avoid SettingWithCopyWarning
    data = df.copy()
    
    # 1. Handle missing values via rolling median
    data['revenue'] = data['revenue'].fillna(data['revenue'].rolling(7, min_periods=1).median())
    
    # 2. Engineer temporal features
    data['day_of_week'] = data['timestamp'].dt.dayofweek
    data['is_weekend'] = np.where(data['day_of_week'] >= 5, 1, 0)
    
    # 3. Scale continuous variables for gradient descent stability
    scaler = StandardScaler()
    data[['revenue', 'user_sessions']] = scaler.fit_transform(data[['revenue', 'user_sessions']])
    
    return data
