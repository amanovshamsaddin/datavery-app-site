import numpy as np
import pandas as pd
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from dataclasses import dataclass, field
from typing import List, Optional

@dataclass
class ResearchRepository:
    name:      str
    domain:    str
    stack:     List[str]
    integrity: bool = True
    status:    str = "PROVISIONED"

    def load_dataset(self, source: str) -> Optional[pd.DataFrame]:
        """Validate and ingest. No fabricated history."""
        if not self.integrity:
            raise ValueError("Integrity check failed.")
        try:
            df = pd.read_csv(source)
            self.status = "ACTIVE"
            return df
        except FileNotFoundError:
            return None  # Dataset pending; system waits.

def architect_system() -> dict:
    repos = [
        ResearchRepository("stats-eda",     "EDA",       ["pandas", "seaborn"]),
        ResearchRepository("db-design",     "Databases", ["postgres"]),
        ResearchRepository("ml-predictive", "ML",        ["sklearn"]),
    ]
    return {
        "operator":    "Shamsaddin Amanov",
        "institution": "Monash University — Clayton",
        "repos":       len(repos),
        "integrity":   all(r.integrity for r in repos),
        "status":      "System ready. Awaiting academic input.",
    }

if __name__ == "__main__":
    report = architect_system()
    for k, v in report.items():
        print(f"  {k:<14}: {v}")

Essential Formulas

Mean (Average)
μ = (1/n) × Σxᵢ

The arithmetic mean is the sum of all values divided by the count. Sensitive to outliers.

Standard Deviation
σ = √(Σ(xᵢ - μ)² / n)

Measures spread around the mean. Higher σ = more dispersed data.

Pearson Correlation
r = Σ(xᵢ-x̄)(yᵢ-ȳ) / (n·σₓ·σᵧ)

Linear relationship between two variables. Range: -1 (inverse) to +1 (perfect).

Linear Regression
ŷ = β₀ + β₁x₁ + ... + βₙxₙ

Predicts a continuous output as a linear combo of inputs with learned weights β.

Sigmoid Function
σ(z) = 1 / (1 + e^(-z))

Maps any real value to [0,1]. Used in logistic regression for probabilities.

MSE Loss
MSE = (1/n) × Σ(yᵢ - ŷᵢ)²

Mean Squared Error penalizes large errors more due to squaring. Most common regression loss.

Bayes' Theorem
P(A|B) = P(B|A)·P(A) / P(B)

Updates probability given new evidence. Foundation of probabilistic ML models.

Shannon Entropy
H(X) = -Σ P(x) · log₂ P(x)

Measures information uncertainty/impurity. Used in decision trees for finding best splits.
